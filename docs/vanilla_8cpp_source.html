31  void Vanilla::iterate() {
32  const size_t n = a.size();
33 
34  for (size_t i = 0; i < n; i++) {
35  a_current[i] = transform.apply_to(a[i]);
36  }
37 
38  // can optimize by just transforming prev centroid if necessary
39  auto a_current_cm = get_centroid(a_current);
40 
41  /*
42  #step
43  Matching Step: match closest points.
44 
45  Sources:
46  https://ieeexplore.ieee.org/stamp/stamp.jsp?tp=&arnumber=4767965
47  https://arxiv.org/pdf/2206.06435.pdf
48  https://web.archive.org/web/20220615080318/https://www.cs.technion.ac.il/~cs236329/tutorials/ICP.pdf
49  https://en.wikipedia.org/wiki/Iterative_closest_point
50  https://courses.cs.duke.edu/spring07/cps296.2/scribe_notes/lecture24.pdf
51  -> use k-d tree
52  */
53  compute_matches();
54 
55  /*
56  #step
57  SVD
58 
59  We compute the SVD of this magical matrix. A proof that this yields the optimal
60  transform R is in the source below.
61 
62  Sources:
63  https://ieeexplore.ieee.org/stamp/stamp.jsp?tp=&arnumber=4767965
64  */
65  Matrix N = Matrix::Zero();
66  for (size_t i = 0; i < n; i++) {
67  N += (a_current[i] - a_current_cm) * (b[matches[i].pair] - b_cm).transpose();
68  }
69  auto svd = N.jacobiSvd(Eigen::ComputeFullU | Eigen::ComputeFullV);
70  const Matrix U = svd.matrixU();
71  Matrix V = svd.matrixV();
72  Matrix R = V * U.transpose();
73 
74  /*
75  #step
76  Reflection Handling
77 
78  SVD may return a reflection instead of a rotation if it's equally good or better.
79  This is exceedingly rare with real data but may happen in very high noise
80  environment with sparse point cloud.
81 
82  In the 2D case, we can always recover a reasonable rotation by negating the last
83  column of V. I do not know if this is the optimal rotation among rotations, but
84  we can probably get an answer to that question with more effort.
85 
86  Sources:
87  https://ieeexplore.ieee.org/stamp/stamp.jsp?tp=&arnumber=4767965
88  */
89  if (R.determinant() < 0) {
90  V = V * Eigen::DiagonalMatrix<double, 2>(1, -1);
91  R = V * U.transpose();
92  }
93 
94  transform.rotation = R * transform.rotation;
95 
96  /*
97  #step
98  Transformation Step: determine optimal transformation.
99 
100  The translation vector is determined by the displacement between
101  the centroids of both point clouds. The rotation matrix is
102  calculated via singular value decomposition.
103 
104  Sources:
105  https://ieeexplore.ieee.org/stamp/stamp.jsp?tp=&arnumber=4767965
106  https://courses.cs.duke.edu/spring07/cps296.2/scribe_notes/lecture24.pdf
107  */
108  transform.translation += b_cm - R * a_current_cm;
109  }

112 void icp::Vanilla::compute_matches() {
113  const size_t n = a.size();
114  const size_t m = b.size();
115 
116  for (size_t i = 0; i < n; i++) {
117  matches[i].cost = std::numeric_limits<double>::infinity();
118  for (size_t j = 0; j < m; j++) {
119  // Point-to-point matching
120  double dist_ij = (b[j] - a_current[i]).squaredNorm();
121 
122  if (dist_ij < matches[i].cost) {
123  matches[i].cost = dist_ij;
124  matches[i].pair = j;
125  }
126  }
127  }
