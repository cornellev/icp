43 void parse_lidar_scan(const char* var, const char* data, void* user_data) {
44  LidarScan* scan = static_cast<LidarScan*>(user_data);
45  if (strcmp(var, "range_min") == 0) {
46  scan->range_min = strtod(data, NULL);
47  } else if (strcmp(var, "range_max") == 0) {
48  scan->range_max = strtod(data, NULL);
49  } else if (strcmp(var, "angle_max") == 0) {
50  scan->angle_max = strtod(data, NULL);
51  Log << "scan->angle_max = " << scan->angle_max << '\n';
52  } else if (strcmp(var, "angle_min") == 0) {
53  scan->angle_min = strtod(data, NULL);
54  Log << "scan->angle_min = " << scan->angle_min << '\n';
55  } else if (strcmp(var, "angle_increment") == 0) {
56  scan->angle_increment = strtod(data, NULL);
57  } else if (isdigit(var[0])) {
58  long index = strtol(var, NULL, 10);
59  double angle = scan->angle_min + index * scan->angle_increment;
60  double range = strtod(data, NULL);
61  if (range >= scan->range_min && range <= scan->range_max) {
62  scan->points.push_back(icp::Vector(100 * range * std::cos(angle),
63  100 * range * std::sin(angle)));
64  // auto last = scan->points.back();
65  // std::cerr << last.x() << ',' << last.y() << '\n';
66  }
67  }

101 void run_benchmark(const char* method, std::unique_ptr<icp::ICP> icp, const LidarScan& source,
102  const LidarScan& destination) {
103  std::cout << "ICP ALGORITHM BENCHMARKING\n";
104  std::cout << "=======================================\n";
105 
106  constexpr size_t N = 50;
107  constexpr size_t burn_in = 0;
108  constexpr double angle_tol = 0.1 * M_PI / 180;
109  constexpr double trans_tol = 0.1;
110 
111  std::cout << "* Method name: " << method << '\n';
112  std::cout << "* Number of trials: " << N << '\n';
113  std::cout << "* Burn-in period: " << burn_in << '\n';
114  std::cout << "* Angle tolerance: " << angle_tol << " rad\n";
115  std::cout << "* Translation tolerance: " << trans_tol << '\n';
116 
117  icp::ICPDriver driver(std::move(icp));
118  driver.set_transform_tolerance(0.1 * M_PI / 180, 0.1);
119 
120  std::vector<double> final_costs;
121  std::vector<size_t> iteration_counts;
122 
123  const auto start = std::chrono::high_resolution_clock::now();
124 
125  for (size_t i = 0; i < N; i++) {
126  auto result = driver.converge(source.points, destination.points, icp::RBTransform());
127  final_costs.push_back(result.cost);
128  iteration_counts.push_back(result.iteration_count);
129  }
130  const auto end = std::chrono::high_resolution_clock::now();
131 
132  const std::chrono::duration<double> diff = end - start;
133 
134  double min_cost = final_costs.front();
135  double max_cost = final_costs.back();
136  double median_cost = final_costs[final_costs.size() / 2];
137  double mean_cost = std::accumulate(final_costs.begin(), final_costs.end(), 0.0)
138  / final_costs.size();
139 
140  std::sort(iteration_counts.begin(), iteration_counts.end());
141 
142  size_t min_iterations = iteration_counts.front();
143  size_t max_iterations = iteration_counts.back();
144  size_t median_iterations = iteration_counts[iteration_counts.size() / 2];
145  double mean_iterations = std::accumulate(iteration_counts.begin(), iteration_counts.end(), 0.0)
146  / iteration_counts.size();
147 
148  std::cout << "* Min cost: " << min_cost << "\n"
149  << "* Max cost: " << max_cost << "\n"
150  << "* Median cost: " << median_cost << "\n"
151  << "* Mean cost: " << mean_cost << "\n";
152  std::cout << "* Min iterations: " << min_iterations << " (real: " << (min_iterations - burn_in)
153  << ")\n"
154  << "* Max iterations: " << max_iterations << " (real: " << (max_iterations - burn_in)
155  << ")\n"
156  << "* Median iterations: " << median_iterations
157  << " (real: " << (median_iterations - burn_in) << ")\n"
158  << "* Mean iterations: " << mean_iterations
159  << " (real: " << (mean_iterations - burn_in) << ")\n";
160  std::cout << "* Average time per invocation: " << (diff.count() / N) << "s\n";
161  std::cout << "* Average time per iteration: " << ((diff.count() / N) / mean_iterations)
162  << "s\n";
163 }
